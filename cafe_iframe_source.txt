function getCheck() {
	var popupCookie = document.cookie;
	var startPoint = popupCookie.indexOf("hidePopupid=");

	if(startPoint != -1){
		var start = startPoint + 12;
		var end = popupCookie.indexOf(";", start);
		if(end == -1) end = popupCookie.length;
		var value = popupCookie.substring(start, end);
		value = decodeURIComponent(value);
		
		var comma = value.length;
		while(comma > 0){
			var comma = value.indexOf("\,");
			var character;
			if(comma == -1){
				character = value;
			}else{
				character = value.substring(0, comma);
				value = value.substring(comma+1, value.length);
			}
			if(document.getElementById(character) != null){

			}
		}
	}

function setHidePopup(closePopup) {
	var popupCookie = document.cookie;
	var startPoint = popupCookie.indexOf("hidePopupid=");
	var closePopupArrayValue = "";
	
	if(startPoint != -1){
		var start = startPoint + 12;
		var end = popupCookie.indexOf(";", start);
		if(end == -1) end = popupCookie.length;
		closePopupArrayValue = popupCookie.substring(start, end);
	}
	if(closePopupArrayValue == ""){
		closePopupArrayValue = closePopup
	}else{
		closePopupArrayValue = closePopupArrayValue + "\," +closePopup;
	}
	
	var expireDate = new Date();
	expireDate.setDate(expireDate.getDate() + 7);
	document.cookie = "hidePopupid="+closePopupArrayValue+"; expires="+expireDate.toGMTString()+";";
}
